Breakout Online!

By this time, you should be quite familiar with Breakout :). Consult the Breakout page for a refresher on the requirements. Here, we simply implement breakout in our javascript framework. 

var ANCHO_DE_LA_ETAPA = 400;
var ALTURA_DE_LA_ETAPA = 600;

var aplicación = new AplicaciónDeGráficos();

/* estado de juego local */
var BRECHA_SUPERIOR = 20; //Separación de la pared superior al ladrillo de abeto.
var BRICK_BRAP = 5; // Brecha entre ladrillos
var BRICKS_ROW = 10; // Número de ladrillos en una fila
var NUM_ROWS = 10; // Número de filas de ladrillos.

var ANCHO_DE_LADRILLO = Mates.piso((ANCHO_DE_LA_ETAPA - BRICK_BRAP*(BRICKS_ROW+2))/BRICKS_ROW);
var LADRILLO_ALTURA = 10;

var RADIO_DE_LA_BOLA = 10;

var PODER_ANCHO = 80;
var ALTURA_DE_PADDLE = 20;

var ladrillosIzquierda = 100; // Número de ladrillos que quedan en la pantalla. si es 0, el jugador gana

var colores = [Color.rojo, Color.amarillo, Color.magenta, Color.verde, Color.azul];

var bola;
var paleta;

/**
 * Creamos todos los elementos de juego aquí:
 * pelota, paleta y ladrillos!
 */
función run() {
    aplicación.añadirLienzo(ANCHO_DE_LA_ETAPA, ALTURA_DE_LA_ETAPA);
    for (var i=0; i < NUM_ROWS; ++i) {
        for (var j=0; j < BRICKS_ROW; ++j) {
            var rect = new GRect(BRICK_BRAP + (i*(ANCHO_DE_LADRILLO+BRICK_BRAP)), 
                                 BRECHA_SUPERIOR + BRICK_BRAP + (j*(LADRILLO_ALTURA+BRICK_BRAP)), 
                                 ANCHO_DE_LADRILLO, LADRILLO_ALTURA);
            rect.establecerColor(colores[Mates.piso(j/2)]);
            añadir(rect);
        }
    }
    
    bola = new CírculoDeG(200,400,RADIO_DE_LA_BOLA);
    bola.vx = Mates.aleatorio()*20-10;
    bola.vy = 6;
    añadir(bola);
    
    paleta = new GRect(180,550,PODER_ANCHO,ALTURA_DE_PADDLE);
    añadir(paleta);
}

/**
 * Detecta cuando la pelota ha golpeado alguna de
 * las paredes. La pelota rebota en los muros norte,
 * este y oeste, e indica una falla si la pelota
 * sale de la parte inferior de la pantalla
 */
función detectarMuros() {
    if (bola.posición.y > ALTURA_DE_LA_ETAPA) {
        var texto = new PIXI.Texto("TÚ PIERDES", {fuente:"50px Arial", llenar:"rojo"});
        texto.posición.x = ANCHO_DE_LA_ETAPA/2 - texto. anchura/2;
        texto.posición.y = ALTURA_DE_LA_ETAPA/2;
        escenario.añadirNiño(texto);
        bola.vx = 0;
        bola.vy = 0;
        return;
    } else if (bola.posición.x < 0 || bola.posición.x > ANCHO_DE_LA_ETAPA) {
        bola.vx = -bola.vx;
    } else if (bola.posición.y <= 0) {
        bola.vy = -bola.vy;   
    }
}

/**
 * ¡Detecta una colisión entre la pelota y otro
 * objeto! Si choca con algún elemento (ladrillo o
 * paleta), revertimos su dirección y. Además, si el
 * colisionador es un ladrillo, eliminamos el
 * ladrillo.
 */
función detectarColisión() {
    
    detectarMuros();
    for (var i=0; i<=RADIO_DE_LA_BOLA; i+=RADIO_DE_LA_BOLA) {
        for (var j=0; j 0 && mousex < ANCHO_DE_LA_ETAPA-PODER_ANCHO) {
        paleta.posición.x = mousex;
    }
    if (ladrillosIzquierda === 0) {
        var texto = new PIXI.Texto("TÚ GANAS", {fuente:"50px Arial", llenar:"rojo"});
        texto.posición.x = ANCHO_DE_LA_ETAPA/2 - texto. anchura/2;
        texto.posición.y = ALTURA_DE_LA_ETAPA/2;
        escenario.añadirNiño(texto);
        bola.vx = 0;
        bola.vy = 0;
    }
}
