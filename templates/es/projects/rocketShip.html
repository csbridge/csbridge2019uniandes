Rockets

Written by Chris Piech

Handouts: ArrayLists

En el clic del mouse disparar un cohete! No hay límite para la cantidad de cohetes que puedes disparar.

Solution

public class RocketPaddle extends ProgramaDeGraficos {
	
	// constantes
	public static final int ALTURA_DE_APLICACION = 600;
	private static final double TAMANO_DE_LA_BOLA = 10;
	private static final int PADLE_Y = ALTURA_DE_APLICACION - 50;
	private static final int PODER_ANCHO = 100;
	private static final int ALTURA_DE_PADDLE = 20;
	
	// la lista de cohetes
	private ArrayList<SOvalo> listaDeCohetes;
	private GRect paleta;
	
	public void run() {
		listaDeCohetes = new ListaDeArreglo();
		paleta = new GRect(0, PADLE_Y, PODER_ANCHO, ALTURA_DE_PADDLE);
		paleta.conjuntoLleno(true);
		añadir(paleta);
		añadirMouseListeners();
		while(true) {
			cohetesAnimados();
			pausa(10);
		}
	}
	
	public void ratónMovido(EventoDelRatón e){
		double x = e.obtenerX() - PODER_ANCHO/2;
		paleta.escogerLocalización(x, PADLE_Y);
	}
	
	public void ratónPresionado(EventoDelRatón e) {
		double x = e.obtenerX();
		double y = PADLE_Y;
		GOval r = new GOval(x, y, TAMAÑO_DE_LA_BOLA, TAMAÑO_DE_LA_BOLA);
		r.conjuntoLleno(true);
		r.establecerColor(Color.AZUL);
		
		// agrega el cohete a la pantalla
		añadir(r);
		
		// agrega el cohete a la lista
		listaDeCohetes.añadir(r);
	}

	private void cohetesAnimados() {
		// bucle sobre la lista hacia atrás para que podamos eliminar de forma segura
		// de la lista.
		for(int i = listaDeCohetes.tamaño() - 1; i >= 0; i--) {
			GOval oval = listaDeCohetes.obtener(i);
			
			// mover el cohete
			oval.moverse(0, -5);
			
			// removerse el cohete?
			if(oval.obtenerY() < 0) {
				retirar(listaDeCohetes.obtener(i));
				listaDeCohetes.retirar(i);
			}
		}
	}
}
