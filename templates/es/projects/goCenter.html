Go to Center

Problem written by Chris and Julia.

This program animates a square that moves until it is in the center of the screen. It starts on the left side.

Solution

public class IrAlCentro extends ProgramaDeGraficos {

    private static final int TAMAÑO = 20;
    private static final int RETRASAR = 100;
	
	public void run() {
		// Dibuja un cuadrado a la izquierda de la pantalla.
		GRect rect = new GRect(TAMAÑO, TAMAÑO);
		double x = 0;
		double y = (obtenerAltura() - rect.obtenerAltura()) / 2;
		rect.conjuntoLleno(true);
		añadir(rect, x, y);
		
		// Mover hasta el centro
		// rect.obtenerX () devuelve la coordenada x del rectángulo
		double maxX = (obtenerAncho() - rect.obtenerAncho()) /2;
		while(rect.obtenerX() < maxX) {
		    // moverse un pixel a la derecha
			rect.moverse(1, 0);
			// pausa de animación
			pausa(RETRASAR);
		}
	}
}
