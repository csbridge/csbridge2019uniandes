Mystery Square

Problem written by a mystery CS106A professor.

Draw a square (100 pixels by 100 pixels) in the center of the screen and change its color every second.

Solution

/**
 * Dibuja un Cuadrado de Misterio que cambia de color.
 * ---------------------------------------------------
 * Realmente hace lo que dice.
 */
public class PlazaDelMisterio extends ProgramaDeGraficos {

    private static final int TAMAÑO = 100;
    private static final int RETRASAR = 1000;

	private static GeneradorAleatorio rg = new GeneradorAleatorio();
	
	public void run() {
		// Dibuja un cuadrado en el centro de la pantalla.
		GRect rect = new GRect(TAMAÑO, TAMAÑO);
		double x = (obtenerAncho() - rect.obtenerAncho()) / 2;
		double y = (obtenerAltura() - rect.obtenerAltura()) / 2;
		rect.conjuntoLleno(true);
		añadir(rect, x, y);
		
		// Cambia su color cada segundo.
		while(true) {
			rect.establecerColor(rg.siguienteColor()); 
			pausa(RETRASAR);
		}
	}
}
