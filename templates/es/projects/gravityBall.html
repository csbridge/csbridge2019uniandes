Gravity Ball

Problem written by Chris.

Hace una bola que rebota como si estuviera bajo la influencia de la gravedad.

Solution

public class BolaDeGravedad extends EsGraphics {

    private static final int TAMANO = 20;
    private static final int RETRASAR = 20;
    
    private static final double GRAVEDAD = 0.5;
    private static final double MOJADURA = 0.7;
	
	public void run() {
		// Dibuja una bola en la esquina superior derecha
		SOval oval = new SOval(0,0,TAMANO, TAMANO);
		oval.cambiarRelleno(true);
		agregar(oval);
		
		// Declarar variables de velocidad.
		double vx = 3;
		double vy = 0;
		
		while(true) {
			// actualizar vy
			vy = vy + GRAVEDAD;
			
			// ¿Debe rebotar la pelota?
			if(oval.darY() > (darAltura() - TAMANO) && vy > 0) {
				vy = vy * -MOJADURA;
			}
			
		    // moverse
			oval.moverse(vx, vy);
			
			// pausa de animación
			pausa(RETRASAR);
		}
	}
}
