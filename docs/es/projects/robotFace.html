Dibuja una cara de robot

Escrito por Eric Roberts. Actualizado por Chris Piech.

Escribe un programa que dibuje esta imagen. Te recomendamos usar las constantes dadas, pero sientete libre de cambiarlas (o recolorear cualquier parte de la cara)

Te damos algunas constantes. Úsalas para no tener que usar números "mágicos" en el programa.

/*El ancho de la cara del robot:*/
private static final int ANCHO_DE_CARA = 300;

/*La altura de la cara del robot:*/
private static final int ALTURA_DE_CARA = 350;

/*El diámetro de cada ojo robot:*/
private static final int DIAMETRO_DE_OJOS = 70;

/*La distancia desde la parte superior de la cabeza hasta la parte superior de los ojos:*/
private static final int COMPENSACION_OJO_Y = 40;

/*El ancho de la boca:*/
private static final int ANCHO_DE_BOCA = 150;

/*La altura de la boca:*/
private static final int ALTURA_DE_BOCA = 30;

/*La distancia desde la parte superior de la cabeza hasta la parte superior de la boca:*/
private static final int COMPENSACION_BOCA_Y = 200;

/*La distancia desde la parte superior de la pantalla hasta la base de la etiqueta:*/
private static final int ETIQUETA_Y = 50;

Solución

/**
 * Dibujar cara
 * ------------
 * Dibuja una impresionante cara de robot, con
 * etiquetas y ojos de diferentes colores. @#$%#@!
 */
public class DibujarCara extends EsGraphics {

	private static final int ANCHO_DE_CARA = 300;
	private static final int ALTURA_DE_CARA = 350;

	/*El ancho o el alto de un ojo:*/
	private static final int DIAMETRO_DE_OJOS = 70;

	/*La distancia desde la cima de la cara hasta la cima de los ojos:*/
	private static final int COMPENSACION_OJO_Y = 40;

	private static final int ANCHO_DE_BOCA = 150;
	private static final int ALTURA_DE_BOCA = 30;

	/*La distancia desde la cima de la cara hasta la cima de la boca:*/
	private static final int COMPENSACION_BOCA_Y = 200;

	/*La distancia desde la cima del canvas hasta el fondo de la etiqueta:*/
	private static final int ETIQUETA_Y = 50;

	public void run() {
		dibujarLaCabeza();
		dibujarOjoIzquierdo();
		dibujarOjoDerecho();
		dibujarBoca();
		dibujarEtiqueta();
	}

	private void dibujarLaCabeza() {
		double x = (darAncho() - ANCHO_DE_CARA) / 2;
		double y = (darAlto() - ALTURA_DE_CARA) / 2;
		SRect cabeza = new SRect(x, y, ANCHO_DE_CARA, ALTURA_DE_CARA);
		cabeza.cambiarRelleno(true);
		cabeza.cambiarColor(Color.ORANGE);
		agregar(cabeza);
	}

	private void dibujarOjoIzquierdo() {
		double espaciamientoDeLosOjos = (ANCHO_DE_CARA - 2 * DIAMETRO_DE_OJOS) / 3;
		double cabezaX = (darAncho() - ANCHO_DE_CARA) / 2;
		double cabezaY = (darAlto() - ALTURA_DE_CARA) / 2;
		double x =  cabezaX + espaciamientoDeLosOjos;
		double y = cabezaY + COMPENSACION_OJO_Y;
		SOvalo ojo = new SOvalo(x, y, DIAMETRO_DE_OJOS, DIAMETRO_DE_OJOS);
		ojo.cambiarRelleno(true);
		ojo.cambiarColor(Color.BLUE);
		agregar(ojo);
	}

	private void dibujarOjoDerecho() {
		double espaciamientoDeLosOjos = (ANCHO_DE_CARA - 2 * DIAMETRO_DE_OJOS) / 3;
		double cabezaX = (darAncho() - ANCHO_DE_CARA) / 2;
		double cabezaY = (darAlto() - ALTURA_DE_CARA) / 2;
		double x =  cabezaX + espaciamientoDeLosOjos * 2 + DIAMETRO_DE_OJOS;
		double y = cabezaY + COMPENSACION_OJO_Y;
		SOvalo ojo = new SOvalo(x, y, DIAMETRO_DE_OJOS, DIAMETRO_DE_OJOS);
		ojo.cambiarRelleno(true);
		ojo.cambiarColor(Color.GREEN);
		agregar(ojo);
	}

	private void dibujarBoca() {
		double cabezaX = (darAncho() - ANCHO_DE_CARA) / 2;
		double cabezaY = (darAlto() - ALTURA_DE_CARA) / 2;
		double x = cabezaX + (ANCHO_DE_CARA - ANCHO_DE_BOCA) / 2;
		double y = cabezaY + COMPENSACION_BOCA_Y;
		SRect boca = new SRect(x, y, ANCHO_DE_BOCA, ALTURA_DE_BOCA);
		boca.cambiarRelleno(true);
		agregar(boca);
	}

	private void dibujarEtiqueta() {
		SEtiqueta etiqueta = new SEtiqueta("Cara de robot");
		etiqueta.cambiarFuente("Courier-50");
		double x = (darAncho() - etiqueta.darAncho()) / 2;
		agregar(etiqueta, x, ETIQUETA_Y);
	}
}
