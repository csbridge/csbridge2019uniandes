Circulo en movimiento

Escrito por Chris Piech

Folletos: Referencia Gráfica

Este programa simplemente dibuja un círculo, luego lo anima moviéndose por la pantalla.

Cada círculo es de 20 píxeles por 20 píxeles.

Solución 

 Clase pública CirculoAlAzar2 extiende ProgramaDeGraficos {

	GeneradorAleatorio privado = nuevo GeneradorAleatorio ();
	public static final int ANCHO_DE_APLICACIÓN = 500;
	public static final int ALTURA_DE_APLICACION = 500;
	final estático privado int TAMAÑO = 20;
	
	public void run () {
		para (int i = 0; i <1000; i ++) {
			int x = rd.siguienteInt (obtenerAncho () - TAMAÑO);
			int y = rd.siguienteInt (obtenerAltura () - TAMAÑO);

			GOval o = new GOval (x, y, TAMAÑO, TAMAÑO);
			o.conjuntoLleno (true);
			o.establecerColor (obtenerColor (x, y));
			añadir (o);
			pausa (2);
		}
	}

	Color privado obtenerColor (int x, int y) {
		if (esVerde (x, y)) {
			vuelve Color.verde;
		}
		vuelve Color.azul;
	}
	
	esVerde booleano privado (int x, int y) {
		devuelve x> y;
	}
}
