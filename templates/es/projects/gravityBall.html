Gravity Ball

Problem written by Chris.

Makes a ball that bounces as if under the influence of gravity.

Solution

public class BolaDeGravedad extends ProgramaDeGraficos {

    private static final int TAMAÑO = 20;
    private static final int RETRASAR = 20;
    
    private static final double GRAVEDAD = 0.5;
    private static final double MOJADURA = 0.7;
	
	public void run() {
		// Dibuja una bola en la esquina superior derecha
		GOval oval = new GOval(0,0,TAMAÑO, TAMAÑO);
		oval.conjuntoLleno(true);
		añadir(oval);
		
		// Declarar variables de velocidad.
		double vx = 3;
		double vy = 0;
		
		while(true) {
			// actualizar vy
			vy = vy + GRAVEDAD;
			
			// ¿Debe rebotar la pelota?
			if(oval.obtenerY() > (obtenerAltura() - TAMAÑO) && vy > 0) {
				vy = vy * -MOJADURA;
			}
			
		    // moverse
			oval.moverse(vx, vy);
			
			// pausa de animación
			pausa(RETRASAR);
		}
	}
}
