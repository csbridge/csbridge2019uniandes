Draw People

Written by Chris, Bryce and Julia

Handouts: Graphics Reference

This demo shows a simple method that draws people with the bottom of their feet at the provided (x, y) position.

private void dibujarPersona(int x, int y)

Solution

public class DibujarPersonas extends ProgramaDeGraficos {

	public void run() {
		dibujarPersona(50, 150);
		dibujarPersona(100, 300);
		dibujarPersona(500, 200);
		dibujarPersona(300, 250);
		dibujarPersona(700, 310);
	}

	private void dibujarPersona(int x, int y) {
		GRect cuerpo = new GRect(20, 80);
		cuerpo.establecerColor(Color.NEGRO);
		cuerpo.conjuntoLleno(true);
		añadir(cuerpo, x - cuerpo.obtenerAncho()/2, y - cuerpo.obtenerAltura());
		
		GRect brazos = new GRect(40, 40);
		brazos.establecerColor(Color.NEGRO);
		brazos.conjuntoLleno(true);
		añadir(brazos, x - brazos.obtenerAncho()/2, y - 60);
		
		GOval cabeza = new GOval(40, 40);
		cabeza.establecerColor(Color.AZUL);
		cabeza.conjuntoLleno(true);
		añadir(cabeza, x - 20, y - 100);
	}
}
